   24   #ifndef _TYPEDEFS_H_
   25   #define _TYPEDEFS_H_
   26   #ifdef __MWERKS__                      //Metrowerk CodeWarrior
   27   #include <stdint.h>
   28   
   29   // Standard typedefs used by header files, based on ISO C standard
   30   typedef volatile int8_t vint8_t;
   31   typedef volatile uint8_t vuint8_t;
   32   typedef volatile int16_t vint16_t;
   33   typedef volatile uint16_t vuint16_t;
   34   typedef volatile int32_t vint32_t;
   35   typedef volatile uint32_t vuint32_t;
   36   
   37   #endif                                 /*__MWERKS__*/
   38   
   39   #ifdef __DCC__                         //Diab
   40   
   41   // This is needed because Diab does not currently have a stdint.h file
   42   typedef signed char int8_t;
   43   typedef unsigned char uint8_t;
   44   typedef volatile signed char vint8_t;
   45   typedef volatile unsigned char vuint8_t;
   46   typedef signed short int16_t;
   47   typedef unsigned short uint16_t;
   48   typedef volatile signed short vint16_t;
   49   typedef volatile unsigned short vuint16_t;
   50   typedef signed int int32_t;
   51   typedef unsigned int uint32_t;
   52   typedef volatile signed int vint32_t;
   53   typedef volatile unsigned int vuint32_t;
   54   
   55   #endif                                 /*__DCC__*/
   56   
   57   #ifdef __ghs__                         //Greenhills
   58   
   59   // This is needed because Greenhills does not currently have a stdint.h file
   60   typedef signed char int8_t;
   61   typedef unsigned char uint8_t;
   62   typedef volatile signed char vint8_t;
   63   typedef volatile unsigned char vuint8_t;
   64   typedef signed short int16_t;
   65   typedef unsigned short uint16_t;
   66   typedef volatile signed short vint16_t;
   67   typedef volatile unsigned short vuint16_t;
   68   typedef signed int int32_t;
   69   typedef unsigned int uint32_t;
   70   typedef volatile signed int vint32_t;
   71   typedef volatile unsigned int vuint32_t;
   72   
   73   #endif                                 /*__ghs__*/
   74   #endif
